# 발주 실행 통합 구현 계획

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 기존 Flask 앱에 발주 실행 기능을 추가하여 바코드+수량 입력 → 발주서 생성 → NAS 저장 → Google Sheets 기록 → 이메일 발송을 HTML 한 곳에서 처리한다.

**Architecture:** 기존 레이어드 아키텍처(domain/application/infrastructure/presentation) 유지. 신제품 등록 시 `통합발주서원본/` 자동 저장 추가. 발주 실행은 `통합발주서원본/*.xlsx`를 파싱해 상품 카탈로그를 구성하고, 원본 파일에 수량을 채워 발주서를 생성한다. PO 코드는 Google Sheets에서 마지막 번호를 읽어 제안하고 로컬 폴더를 fallback으로 사용한다.

**Tech Stack:** Flask 3.0, openpyxl 3.1, google-api-python-client 2.x, google-auth 2.x, Python smtplib (표준 라이브러리), Docker on Synology DS218+

## Google Sheets 사전 설정 (Task 1 전에 수동으로)

1. [Google Cloud Console](https://console.cloud.google.com/) → 새 프로젝트 생성
2. "Google Sheets API" 활성화
3. "서비스 계정" 생성 → JSON 키 다운로드 → `google_service_account.json`으로 저장
4. 엔대시 Google Sheet 열기 → 공유 → 서비스 계정 이메일 추가 (편집자 권한)
5. Sheet URL에서 `SHEET_ID` 복사: `https://docs.google.com/spreadsheets/d/{SHEET_ID}/edit`
6. `google_service_account.json`을 Synology에 업로드: `/volume1/secrets/google_service_account.json`

`infrastructure/발주서원본_파서.py`:

```python
"""
발주서원본_파서 - 통합발주서원본/*.xlsx → 바코드별 상품_정보 dict
"""
from pathlib import Path
import openpyxl
from domain.entities.상품_정보 import 상품_정보


class 발주서원본_파서:

    def __init__(self, 원본_디렉토리: Path):
        self._디렉토리 = 원본_디렉토리

    def 카탈로그_로드(self) -> dict[str, 상품_정보]:
        """바코드 → 상품_정보 dict 반환. 중복 바코드는 나중 파일이 덮어씀."""
        카탈로그: dict[str, 상품_정보] = {}

        for xlsx_파일 in sorted(self._디렉토리.glob("*.xlsx")):
            try:
                self._파일_파싱(xlsx_파일, 카탈로그)
            except Exception as e:
                print(f"[발주서원본_파서] 파싱 실패 {xlsx_파일.name}: {e}")

        return 카탈로그

    def _파일_파싱(self, 파일: Path, 카탈로그: dict) -> None:
        제조사, 브랜드 = self._파일명_파싱(파일.stem)

        wb = openpyxl.load_workbook(파일, read_only=True, data_only=True)
        ws = wb.active

        # 행 7부터 B(번호)가 빌 때까지 읽음
        for row in ws.iter_rows(min_row=7, values_only=True):
            if len(row) < 12:
                break
            번호_val = row[1]   # B
            if not 번호_val:
                break

            바코드_val = row[8]  # I
            if not 바코드_val:
                continue

            바코드_str = str(바코드_val).strip()
            if not 바코드_str:
                continue

            카탈로그[바코드_str] = 상품_정보(
                바코드=바코드_str,
                이름=str(row[7] or ""),          # H
                브랜드=브랜드,
                제조사=제조사,
                원가=float(row[9] or 0),          # J
                수출_usd=float(row[4] or 0),       # E
                moq=str(row[5] or ""),             # F
                item=str(row[2] or ""),            # C
                model_no=str(row[3] or ""),        # D
                번호=int(번호_val),
                원본파일=파일,
            )

        wb.close()

    @staticmethod
    def _파일명_파싱(stem: str) -> tuple[str, str]:
        """
        '富坤电器_NUART 테스트 제품(NUART전용)' → ('富坤电器', 'NUART')
        '_' 없으면 ('', stem) 반환.
        """
        if "_" not in stem:
            return "", stem
        제조사, 나머지 = stem.split("_", 1)
        브랜드 = 나머지.split(" ")[0]
        return 제조사, 브랜드
```

```python
import config
# (기존 import 유지)

@bp.route("/api/발주서/생성", methods=["POST"])
def 발주서_생성():
    result = _유스케이스.실행(request.form, request.files)

    if not result["success"]:
        return jsonify(result), 400

    # ── NAS 원본 폴더에 자동 저장 ─────────────────────────────
    try:
        저장_경로 = config.통합발주서원본_DIR / result["파일명"]
        config.통합발주서원본_DIR.mkdir(parents=True, exist_ok=True)
        result["버퍼"].seek(0)
        저장_경로.write_bytes(result["버퍼"].read())
        result["버퍼"].seek(0)  # 다운로드용으로 되감기
    except Exception as e:
        # 저장 실패는 경고만 출력. 다운로드는 계속 진행.
        print(f"[routes] NAS 저장 실패: {e}")

    return send_file(
        result["버퍼"],
        as_attachment=True,
        download_name=result["파일명"],
        mimetype="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
    )
```

```python
"""
발주_코드_서비스 - PO 코드 파싱/포맷/다음번호 계산
"""
import re
from pathlib import Path


class 발주_코드_서비스:

    @staticmethod
    def 파싱(코드: str) -> tuple[str, int]:
        """'NF(694)' → ('NF', 694). 형식 불일치 시 ValueError."""
        m = re.match(r'^([A-Za-z]+)\((\d+)\)$', 코드.strip())
        if not m:
            raise ValueError(f"잘못된 PO 코드 형식: {코드!r}")
        return m.group(1), int(m.group(2))

    @staticmethod
    def 포맷(prefix: str, 번호: int) -> str:
        """('NF', 695) → 'NF(695)'"""
        return f"{prefix}({번호})"

    @staticmethod
    def 유효성_검사(코드: str) -> bool:
        return bool(re.match(r'^[A-Za-z]+\(\d+\)$', 코드.strip()))

    @staticmethod
    def 다음_번호_로컬(발주서_디렉토리: Path, prefix: str) -> int:
        """
        발주서_디렉토리의 정기발주/ + 신기종/ 폴더를 스캔해
        prefix(xxx).xlsx 파일 중 최대 xxx + 1 반환.
        없으면 1 반환.
        """
        max_num = 0
        pattern = re.compile(rf'^{re.escape(prefix)}\((\d+)\)\.xlsx$')
        for sub in ("정기발주", "신기종"):
            folder = 발주서_디렉토리 / sub
            if not folder.exists():
                continue
            for f in folder.glob("*.xlsx"):
                m = pattern.match(f.name)
                if m:
                    max_num = max(max_num, int(m.group(1)))
        return max_num + 1
```

```python
"""
Google Sheets 저장소 - 엔대시 시트 조회/기록
"""
import re
from datetime import date
from google.oauth2 import service_account
from googleapiclient.discovery import build


_SCOPES = ["https://www.googleapis.com/auth/spreadsheets"]


class GoogleSheets저장소:

    def __init__(self, 서비스계정_경로: str, 스프레드시트_id: str, 시트명: str):
        self._스프레드시트_id = 스프레드시트_id
        self._시트명 = 시트명
        creds = service_account.Credentials.from_service_account_file(
            서비스계정_경로, scopes=_SCOPES
        )
        self._service = build("sheets", "v4", credentials=creds, cache_discovery=False)

    def 마지막_po번호_조회(self, prefix: str) -> int:
        """
        시트에서 발주번호(C열 기준) 중 prefix(xxx) 최대값 반환.
        없거나 오류 시 0 반환.
        """
        try:
            result = (
                self._service.spreadsheets()
                .values()
                .get(
                    spreadsheetId=self._스프레드시트_id,
                    range=f"{self._시트명}!C:C",
                )
                .execute()
            )
            rows = result.get("values", [])
        except Exception as e:
            print(f"[GoogleSheets] 조회 실패: {e}")
            return 0

        pattern = re.compile(rf'^{re.escape(prefix)}\((\d+)\)$')
        max_num = 0
        for row in rows:
            if row and pattern.match(str(row[0]).strip()):
                max_num = max(max_num, int(pattern.match(row[0].strip()).group(1)))
        return max_num

    def 발주_기록(
        self,
        바코드: str,
        po코드: str,
        수량: int,
        제품명: str,
        제조사: str,
    ) -> None:
        """엔대시 시트에 한 행 추가. 발주일자는 오늘 날짜 자동."""
        오늘 = date.today()
        발주일자 = f"{오늘.year}. {오늘.month}. {오늘.day}"

        # 컬럼 순서: 발주일자 | 바코드 | 발주번호 | 쿠팡수량 | 일반수량 | 제품명 | 비고 | 중국출발 | 누아트입고 | 14일경과 | 제조사
        row = [발주일자, 바코드, po코드, 수량, "", 제품명, "", "", "", "", 제조사]

        self._service.spreadsheets().values().append(
            spreadsheetId=self._스프레드시트_id,
            range=f"{self._시트명}!A:K",
            valueInputOption="USER_ENTERED",
            body={"values": [row]},
        ).execute()
```

```python
"""
이메일_저장소 - SMTP 발주서 첨부 이메일 발송
"""
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email.mime.text import MIMEText
from email import encoders
from pathlib import Path


class 이메일_저장소:

    def __init__(
        self,
        smtp_서버: str,
        smtp_포트: int,
        발신자: str,
        비밀번호: str,
        수신자: str,
    ):
        self._smtp_서버 = smtp_서버
        self._smtp_포트 = smtp_포트
        self._발신자 = 발신자
        self._비밀번호 = 비밀번호
        self._수신자 = 수신자

    def 발주서_발송(
        self,
        파일_경로: Path,
        po코드: str,
        제조사: str,
        제품명: str,
    ) -> None:
        """발주서 xlsx를 첨부하여 수신자에게 이메일 발송."""
        msg = MIMEMultipart()
        msg["From"] = self._발신자
        msg["To"] = self._수신자
        msg["Subject"] = f"발주서 [{po코드}] {제조사} - {제품명}"

        본문 = (
            f"안녕하세요,\n\n"
            f"발주서를 첨부드립니다.\n\n"
            f"발주번호: {po코드}\n"
            f"제조사: {제조사}\n"
            f"제품명: {제품명}\n\n"
            f"감사합니다."
        )
        msg.attach(MIMEText(본문, "plain", "utf-8"))

        with open(파일_경로, "rb") as f:
            part = MIMEBase("application", "octet-stream")
            part.set_payload(f.read())
            encoders.encode_base64(part)
            # RFC 5987 방식으로 한글 파일명 인코딩
            part.add_header(
                "Content-Disposition",
                f"attachment",
                filename=("utf-8", "", 파일_경로.name),
            )
            msg.attach(part)

        with smtplib.SMTP(self._smtp_서버, self._smtp_포트) as server:
            server.ehlo()
            server.starttls()
            server.login(self._발신자, self._비밀번호)
            server.send_message(msg)
```

```python
"""
발주서_실행_저장소 - 원본 발주서에 수량을 채워 발주서 폴더에 저장
"""
import shutil
from pathlib import Path
import openpyxl


class 발주서_실행_저장소:

    def __init__(self, 발주서_디렉토리: Path):
        self._발주서_디렉토리 = 발주서_디렉토리

    def 발주서_생성(
        self,
        원본_경로: Path,
        출력_경로: Path,
        바코드_수량: dict[str, int],
        한국_요청_도착일: str = "",
    ) -> Path:
        """
        원본 발주서를 복사하고 바코드에 해당하는 수량을 채워 저장.
        Returns: 저장된 파일 경로
        """
        출력_경로.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(원본_경로, 출력_경로)

        wb = openpyxl.load_workbook(출력_경로)
        ws = wb.active

        if 한국_요청_도착일:
            ws["G4"] = 한국_요청_도착일

        # 행 7부터 B(번호)가 빌 때까지 수량 업데이트
        # 열 인덱스: B=2, I=9(바코드), J=10(원가), K=11(수량), L=12(합계)
        for row_idx in range(7, ws.max_row + 1):
            번호_셀 = ws.cell(row=row_idx, column=2)
            if not 번호_셀.value:
                break

            바코드_셀 = ws.cell(row=row_idx, column=9)
            바코드_str = str(바코드_셀.value or "").strip()

            if 바코드_str in 바코드_수량:
                원가 = float(ws.cell(row=row_idx, column=10).value or 0)
                수량 = 바코드_수량[바코드_str]
                ws.cell(row=row_idx, column=11).value = 수량
                ws.cell(row=row_idx, column=12).value = round(원가 * 수량, 2)

        wb.save(출력_경로)
        return 출력_경로

    def 출력_경로_계산(self, 발주_유형: str, po코드: str) -> Path:
        """'정기발주', 'NF(695)' → 발주서_DIR/정기발주/NF(695).xlsx"""
        return self._발주서_디렉토리 / 발주_유형 / f"{po코드}.xlsx"
```

`application/발주_실행_유스케이스.py`:

```python
"""
발주_실행_유스케이스 - 발주 조회 및 실행 흐름 조율
"""
from pathlib import Path
from dataclasses import dataclass, field
from typing import Optional

import config
from infrastructure.발주서원본_파서 import 발주서원본_파서
from infrastructure.발주서_실행_저장소 import 발주서_실행_저장소
from infrastructure.google_sheets_저장소 import GoogleSheets저장소
from infrastructure.이메일_저장소 import 이메일_저장소
from domain.services.발주_코드_서비스 import 발주_코드_서비스


@dataclass
class 조회_항목:
    바코드: str
    이름: str
    원가: float
    수량: int


@dataclass
class 조회_그룹:
    제조사: str
    브랜드: str
    원본파일: str          # 파일명만 (경로 제외)
    po_prefix: str
    po_번호_제안: int
    items: list[조회_항목] = field(default_factory=list)


@dataclass
class 실행_그룹:
    원본파일: str          # 파일명만
    po코드: str
    items: list[dict]      # [{"barcode": "...", "quantity": 500}, ...]


@dataclass
class 실행_결과:
    po코드: str
    파일_경로: str
    이메일_발송: bool
    sheets_기록: bool


class 발주_실행_유스케이스:

    def __init__(self):
        self._파서 = 발주서원본_파서(config.통합발주서원본_DIR)
        self._저장소 = 발주서_실행_저장소(config.발주서_DIR)
        self._sheets: Optional[GoogleSheets저장소] = self._sheets_연결()
        self._이메일: Optional[이메일_저장소] = self._이메일_연결()

    # ── 상품 조회 ────────────────────────────────────────────────

    def 조회(self, items: list[dict]) -> dict:
        """
        items: [{"barcode": "...", "quantity": 500}, ...]
        Returns: {"success": True, "groups": [조회_그룹, ...], "미확인": [...]}
        """
        카탈로그 = self._파서.카탈로그_로드()
        그룹_맵: dict[str, 조회_그룹] = {}
        미확인 = []

        for item in items:
            바코드 = str(item.get("barcode", "")).strip()
            수량 = int(item.get("quantity", 0))

            if 바코드 not in 카탈로그:
                미확인.append(바코드)
                continue

            상품 = 카탈로그[바코드]
            키 = 상품.원본파일.name

            if 키 not in 그룹_맵:
                prefix = config.BRAND_PO_PREFIX.get(상품.브랜드, 상품.브랜드[:2].upper())
                po_번호 = self._다음_po번호(prefix)
                그룹_맵[키] = 조회_그룹(
                    제조사=상품.제조사,
                    브랜드=상품.브랜드,
                    원본파일=키,
                    po_prefix=prefix,
                    po_번호_제안=po_번호,
                )

            그룹_맵[키].items.append(
                조회_항목(바코드=바코드, 이름=상품.이름, 원가=상품.원가, 수량=수량)
            )

        return {
            "success": True,
            "groups": [
                {
                    "제조사": g.제조사,
                    "브랜드": g.브랜드,
                    "원본파일": g.원본파일,
                    "po_prefix": g.po_prefix,
                    "po_번호_제안": g.po_번호_제안,
                    "items": [
                        {"barcode": i.바코드, "이름": i.이름, "원가": i.원가, "수량": i.수량}
                        for i in g.items
                    ],
                }
                for g in 그룹_맵.values()
            ],
            "미확인": 미확인,
        }

    # ── 발주 실행 ────────────────────────────────────────────────

    def 실행(self, payload: dict) -> dict:
        """
        payload:
          발주_유형: "정기발주" | "신기종"
          한국_요청_도착일: "2026. 4. 30" (optional)
          groups: [{원본파일, po코드, items: [{barcode, quantity}]}]
        Returns: {"success": True, "results": [...]}
        """
        발주_유형 = payload.get("발주_유형", "정기발주")
        한국_요청_도착일 = payload.get("한국_요청_도착일", "")
        결과_목록 = []

        for 그룹 in payload.get("groups", []):
            원본파일 = 그룹["원본파일"]
            po코드 = 그룹["po코드"]
            items = 그룹["items"]

            if not 발주_코드_서비스.유효성_검사(po코드):
                return {"success": False, "error": f"잘못된 PO 코드: {po코드}"}

            # 발주서 생성 및 저장
            원본_경로 = config.통합발주서원본_DIR / 원본파일
            출력_경로 = self._저장소.출력_경로_계산(발주_유형, po코드)
            바코드_수량 = {str(i["barcode"]): int(i["quantity"]) for i in items}

            self._저장소.발주서_생성(
                원본_경로=원본_경로,
                출력_경로=출력_경로,
                바코드_수량=바코드_수량,
                한국_요청_도착일=한국_요청_도착일,
            )

            # 제조사/제품명 파악 (카탈로그에서 첫 번째 아이템 기준)
            카탈로그 = self._파서.카탈로그_로드()
            첫_바코드 = str(items[0]["barcode"])
            제조사 = 카탈로그.get(첫_바코드, None)
            제조사명 = 제조사.제조사 if 제조사 else ""
            제품명_대표 = 카탈로그.get(첫_바코드, None)
            제품명 = 제품명_대표.이름 if 제품명_대표 else 원본파일

            # Google Sheets 기록 (각 바코드별 한 행)
            sheets_ok = True
            if self._sheets:
                try:
                    for item in items:
                        바코드 = str(item["barcode"])
                        상품명 = 카탈로그.get(바코드)
                        self._sheets.발주_기록(
                            바코드=바코드,
                            po코드=po코드,
                            수량=int(item["quantity"]),
                            제품명=상품명.이름 if 상품명 else "",
                            제조사=제조사명,
                        )
                except Exception as e:
                    print(f"[발주_실행] Sheets 기록 실패: {e}")
                    sheets_ok = False
            else:
                sheets_ok = False

            # 이메일 발송
            이메일_ok = True
            if self._이메일:
                try:
                    self._이메일.발주서_발송(
                        파일_경로=출력_경로,
                        po코드=po코드,
                        제조사=제조사명,
                        제품명=제품명,
                    )
                except Exception as e:
                    print(f"[발주_실행] 이메일 발송 실패: {e}")
                    이메일_ok = False
            else:
                이메일_ok = False

            결과_목록.append({
                "po코드": po코드,
                "파일": str(출력_경로.relative_to(config.발주서_DIR)),
                "이메일_발송": 이메일_ok,
                "sheets_기록": sheets_ok,
            })

        return {"success": True, "results": 결과_목록}

    # ── 내부 헬퍼 ────────────────────────────────────────────────

    def _다음_po번호(self, prefix: str) -> int:
        """Google Sheets → fallback: 로컬 폴더 스캔"""
        if self._sheets:
            try:
                n = self._sheets.마지막_po번호_조회(prefix)
                if n > 0:
                    return n + 1
            except Exception as e:
                print(f"[발주_실행] Sheets PO번호 조회 실패: {e}")
        return 발주_코드_서비스.다음_번호_로컬(config.발주서_DIR, prefix)

    def _sheets_연결(self) -> Optional[GoogleSheets저장소]:
        if not config.GOOGLE_SERVICE_ACCOUNT_JSON or not config.GOOGLE_SHEET_ID:
            return None
        try:
            return GoogleSheets저장소(
                config.GOOGLE_SERVICE_ACCOUNT_JSON,
                config.GOOGLE_SHEET_ID,
                config.GOOGLE_SHEET_NAME,
            )
        except Exception as e:
            print(f"[발주_실행] Google Sheets 연결 실패: {e}")
            return None

    def _이메일_연결(self) -> Optional[이메일_저장소]:
        if not config.SMTP_발신자 or not config.SMTP_비밀번호 or not config.SMTP_수신자:
            return None
        return 이메일_저장소(
            config.SMTP_서버,
            config.SMTP_포트,
            config.SMTP_발신자,
            config.SMTP_비밀번호,
            config.SMTP_수신자,
        )
```

```html
    ).join("");

    container.innerHTML += `

  });
}

async function 발주실행() {
  const 발주유형 = document.getElementById("발주유형").value;
  const 도착일_raw = document.getElementById("도착일").value;
  let 한국_요청_도착일 = "";
  if (도착일_raw) {
    const d = new Date(도착일_raw);
    한국_요청_도착일 = `${d.getFullYear()}. ${d.getMonth()+1}. ${d.getDate()}`;
  }

  const groups = 그룹_데이터.map((g, gi) => ({
    원본파일: g.원본파일,
    po코드: document.getElementById(`po_${gi}`).value.trim(),
    items: g.items.map(i => ({ barcode: i.barcode, quantity: i.수량 })),
  }));

  const payload = { 발주_유형: 발주유형, 한국_요청_도착일, groups };

  const res = await fetch("/api/발주/실행", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(payload),
  });
  const data = await res.json();

  결과_렌더링(data);
  document.getElementById("결과영역").style.display = "block";
  document.getElementById("결과영역").scrollIntoView({ behavior: "smooth" });
}

function 결과_렌더링(data) {
  const el = document.getElementById("결과내용");
  if (!data.success) {
    el.innerHTML = `<div class="alert alert-danger">${data.error}</div>`;
    return;
  }
  const rows = data.results.map(r => `
    <tr>
      <td><code>${r.po코드}</code></td>
      <td>${r.파일}</td>
      <td class="${r.이메일_발송 ? 'result-ok' : 'result-warn'}">${r.이메일_발송 ? "발송 완료" : "실패/미설정"}</td>
      <td class="${r.sheets_기록 ? 'result-ok' : 'result-warn'}">${r.sheets_기록 ? "기록 완료" : "실패/미설정"}</td>
    </tr>`).join("");

  el.innerHTML = `
    <table class="table">
      <thead><tr><th>PO 코드</th><th>저장 경로</th><th>이메일</th><th>Google Sheets</th></tr></thead>
      <tbody>${rows}</tbody>
    </table>`;
}
```
