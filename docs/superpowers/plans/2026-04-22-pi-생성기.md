```python
"""
계좌 DB 저장소 - 제조사별 계좌정보 CRUD + 구글시트 숨김탭 백업
"""
from __future__ import annotations
import json
import re
import uuid
from pathlib import Path
from typing import Optional

import config


class 계좌저장소:

    _SHEETS_탭 = "계좌DB_백업"

    def __init__(self, sheets=None):
        self._파일 = config.계좌DB_파일
        self._sheets = sheets
        self._파일.parent.mkdir(parents=True, exist_ok=True)
        config.업체도장_DIR.mkdir(parents=True, exist_ok=True)

    # ── CRUD ──────────────────────────────────────────────────────

    def 전체_조회(self) -> dict:
        if not self._파일.exists():
            return {}
        return json.loads(self._파일.read_text(encoding="utf-8"))

    def 단건_조회(self, 회사id: str) -> Optional[dict]:
        return self.전체_조회().get(회사id)

    def 저장(self, 계좌: dict) -> str:
        """계좌 추가/수정. 반환: 회사id"""
        db = self.전체_조회()
        회사id = 계좌.get("id") or self._id_생성(계좌.get("회사명_영문", ""))
        계좌["id"] = 회사id
        db[회사id] = 계좌
        self._파일.write_text(json.dumps(db, ensure_ascii=False, indent=2), encoding="utf-8")
        self._구글시트_백업(db)
        return 회사id

    def 삭제(self, 회사id: str) -> bool:
        db = self.전체_조회()
        if 회사id not in db:
            return False
        # 도장 이미지 삭제
        도장 = db[회사id].get("도장_파일", "")
        if 도장:
            p = config.BASE_DIR / "assets" / 도장
            if p.exists():
                p.unlink()
        del db[회사id]
        self._파일.write_text(json.dumps(db, ensure_ascii=False, indent=2), encoding="utf-8")
        self._구글시트_백업(db)
        return True

    def 도장_저장(self, 회사id: str, 이미지_바이트: bytes, 확장자: str = ".png") -> str:
        """업체 도장 이미지 저장. 반환: 상대경로"""
        경로 = config.업체도장_DIR / f"{회사id}{확장자}"
        경로.write_bytes(이미지_바이트)
        상대경로 = f"업체도장/{회사id}{확장자}"
        db = self.전체_조회()
        if 회사id in db:
            db[회사id]["도장_파일"] = 상대경로
            self._파일.write_text(json.dumps(db, ensure_ascii=False, indent=2), encoding="utf-8")
        return 상대경로

    # ── 텍스트 파싱 ───────────────────────────────────────────────

    @staticmethod
    def 텍스트_파싱(text: str) -> dict:
        """
        PI 하단 Bank information 블록 텍스트에서 계좌 정보 추출.
        예:
          County / City : China / Cixi
          Beneficiary Account Number : 376670899761
          Beneficiary Name  : Cixi Hongcheng...
          SWIFT/BIC code :  BKCHCNBJ92A
          Bank name : BANK OF CHINA...
          Bank Address : CIXI CITY...
          Branch Code : 92A
        """
        def _find(pattern: str) -> str:
            m = re.search(pattern, text, re.IGNORECASE)
            return m.group(1).strip() if m else ""

        결과 = {
            "account_number":     _find(r"Beneficiary Account Number\s*:\s*(.+)"),
            "beneficiary_name":   _find(r"Beneficiary Name\s*:\s*(.+)"),
            "beneficiary_address":_find(r"Beneficiary Address\s*:\s*(.+)"),
            "swift":              _find(r"SWIFT/BIC\s*(?:code)?\s*:\s*(\S+)"),
            "bank_name":          _find(r"Bank name\s*:\s*(.+)"),
            "bank_address":       _find(r"Bank Address\s*:\s*(.+)"),
            "branch_code":        _find(r"Branch Code\s*:\s*(\S+)"),
        }
        # County / City : China / Cixi
        cc = _find(r"County\s*/\s*City\s*:\s*(.+)")
        if "/" in cc:
            parts = [p.strip() for p in cc.split("/", 1)]
            결과["country"] = parts[0]
            결과["city"] = parts[1]
        # beneficiary_name을 회사명으로 사용
        결과["회사명_영문"] = 결과.get("beneficiary_name", "")
        return 결과

    @staticmethod
    def 스크린샷_파싱(이미지_바이트: bytes) -> dict:
        """Claude Vision API로 스크린샷에서 계좌 정보 추출."""
        if not config.ANTHROPIC_API_KEY:
            raise ValueError("ANTHROPIC_API_KEY가 설정되지 않았습니다.")
        import anthropic, base64
        client = anthropic.Anthropic(api_key=config.ANTHROPIC_API_KEY)
        img_b64 = base64.standard_b64encode(이미지_바이트).decode()
        msg = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=1024,
            messages=[{
                "role": "user",
                "content": [
                    {
                        "type": "image",
                        "source": {"type": "base64", "media_type": "image/png", "data": img_b64},
                    },
                    {
                        "type": "text",
                        "text": (
                            "이 이미지에서 은행 계좌 정보를 추출해서 아래 JSON 형식으로만 응답해줘. "
                            "없는 필드는 빈 문자열로.\n"
                            '{"account_number":"","beneficiary_name":"","beneficiary_address":"",'
                            '"swift":"","bank_name":"","bank_address":"","branch_code":"",'
                            '"country":"","city":"","회사명_영문":""}'
                        ),
                    },
                ],
            }],
        )
        raw = msg.content[0].text.strip()
        # JSON 블록 추출
        m = re.search(r'\{.*\}', raw, re.DOTALL)
        if not m:
            raise ValueError(f"JSON 파싱 실패: {raw[:200]}")
        return json.loads(m.group())

    # ── 구글시트 백업 ─────────────────────────────────────────────

    def _구글시트_백업(self, db: dict) -> None:
        if not self._sheets:
            return
        try:
            rows = [["id","회사명_영문","회사명_중문","country","city",
                     "account_number","beneficiary_name","beneficiary_address",
                     "swift","bank_name","bank_address","branch_code","도장_파일"]]
            for 계좌 in db.values():
                rows.append([
                    계좌.get("id",""), 계좌.get("회사명_영문",""), 계좌.get("회사명_중문",""),
                    계좌.get("country",""), 계좌.get("city",""), 계좌.get("account_number",""),
                    계좌.get("beneficiary_name",""), 계좌.get("beneficiary_address",""),
                    계좌.get("swift",""), 계좌.get("bank_name",""), 계좌.get("bank_address",""),
                    계좌.get("branch_code",""), 계좌.get("도장_파일",""),
                ])
            self._sheets._구글시트_탭_덮어쓰기(self._SHEETS_탭, rows)
        except Exception as e:
            print(f"[계좌저장소] 구글시트 백업 실패: {e}")

    @staticmethod
    def _id_생성(회사명: str) -> str:
        slug = re.sub(r'[^a-zA-Z0-9]+', '-', 회사명.lower()).strip('-')
        return slug[:30] or str(uuid.uuid4())[:8]
```

- [ ] **Step 2: GoogleSheets저장소에 탭 덮어쓰기 메서드 추가**

```python
def _구글시트_탭_덮어쓰기(self, 탭명: str, rows: list[list]) -> None:
    """탭 전체를 rows로 덮어씀. 탭 없으면 생성 후 숨김 처리."""
    meta = self._service.spreadsheets().get(
        spreadsheetId=self._스프레드시트_id
    ).execute()
    existing = {s["properties"]["title"]: s["properties"]["sheetId"]
                for s in meta.get("sheets", [])}

    if 탭명 not in existing:
        res = self._service.spreadsheets().batchUpdate(
            spreadsheetId=self._스프레드시트_id,
            body={"requests": [{"addSheet": {"properties": {
                "title": 탭명, "hidden": True
            }}}]},
        ).execute()
        sheet_id = res["replies"][0]["addSheet"]["properties"]["sheetId"]
    else:
        sheet_id = existing[탭명]
        # 기존 탭 숨김 처리 (이미 숨겨져 있어도 무방)
        self._service.spreadsheets().batchUpdate(
            spreadsheetId=self._스프레드시트_id,
            body={"requests": [{"updateSheetProperties": {
                "properties": {"sheetId": sheet_id, "hidden": True},
                "fields": "hidden",
            }}]},
        ).execute()
        # 기존 내용 전체 삭제
        self._service.spreadsheets().values().clear(
            spreadsheetId=self._스프레드시트_id,
            range=f"'{탭명}'!A:ZZ",
        ).execute()

    self._service.spreadsheets().values().update(
        spreadsheetId=self._스프레드시트_id,
        range=f"'{탭명}'!A1",
        valueInputOption="USER_ENTERED",
        body={"values": rows},
    ).execute()
```

```python
"""
PI (Proforma Invoice) xlsx 생성기
템플릿 복사 → 셀 채우기 → 도장 삽입
"""
from __future__ import annotations

from pathlib import Path

import config


class PI생성기:

    def __init__(self):
        self._템플릿 = config.PI_템플릿
        config.PI_DIR.mkdir(parents=True, exist_ok=True)

    def 생성(
        self,
        po코드: str,
        제조사_계좌: dict,
        제품목록: list[dict],
        디포짓_목록: list[dict],
        담당자: str,
        pi_번호: Optional[str] = None,
    ) -> Path:
        """
        PI xlsx 생성 후 파일 경로 반환.

        제품목록 item: {"po_no": str, "brand": str, "description": str,
                        "product_name_kr": str, "barcode": str,
                        "qty": int, "unit_price": float, "already_paid": float}
        디포짓_목록 item: {"label": "1st Deposit", "pct": 0.3, "amount": 14100.0,
                           "예정일": "2026-04-30"}  # amount 또는 pct 중 하나만 필수
        """
        if not self._템플릿.exists():
            raise FileNotFoundError(f"PI 템플릿 없음: {self._템플릿}")

        오늘 = date.today()
        if pi_번호 is None:
            pi_번호 = f"J{오늘.strftime('%Y%m%d')}_{po코드}"

        출력_파일명 = f"{오늘.strftime('%y%m%d')}_{po코드}_{제조사_계좌.get('회사명_영문','')[:20]}.xlsx"
        출력_경로 = config.PI_DIR / po코드
        출력_경로.mkdir(parents=True, exist_ok=True)
        출력_파일 = 출력_경로 / 출력_파일명

        shutil.copy2(self._템플릿, 출력_파일)
        wb = openpyxl.load_workbook(출력_파일)
        ws = wb.active

        총액 = sum(
            int(p["qty"]) * float(p.get("unit_price", 0))
            for p in 제품목록
        )
        기납부액 = sum(float(p.get("already_paid", 0)) for p in 제품목록)

        self._fill_header(ws, 오늘, pi_번호, 제조사_계좌)
        self._fill_products(ws, 제품목록, 총액, 기납부액)
        self._fill_deposits(ws, 디포짓_목록, 총액)
        self._fill_bank_info(ws, 제조사_계좌)
        self._insert_stamps(ws, 제조사_계좌)

        wb.save(출력_파일)
        return 출력_파일

    # ── 내부 메서드 ───────────────────────────────────────────────

    def _find_cell(self, ws, keyword: str):
        """ws 전체에서 keyword를 포함하는 첫 번째 셀 반환."""
        for row in ws.iter_rows():
            for cell in row:
                if cell.value and keyword.lower() in str(cell.value).lower():
                    return cell
        return None

    def _fill_header(self, ws, 오늘: date, pi_번호: str, 계좌: dict):
        # 공급업체명 (A1)
        c = self._find_cell(ws, "공급자 영문명")
        if c is None:
            c = ws.cell(1, 1)
        c.value = 계좌.get("회사명_영문", "")

        # Date
        date_label = self._find_cell(ws, "Date:")
        if date_label:
            ws.cell(date_label.row, date_label.column + 1).value = 오늘.strftime("%Y-%m-%d")

        # PI No.
        pino_label = self._find_cell(ws, "PI No.")
        if pino_label:
            ws.cell(pino_label.row, pino_label.column + 1).value = pi_번호

    def _fill_products(self, ws, 제품목록: list[dict], 총액: float, 기납부액: float):
        """No. 헤더 행 아래 제품 행을 찾아 채움."""
        no_cell = self._find_cell(ws, "No.")
        if no_cell is None:
            return
        header_row = no_cell.row
        # 기존 제품 행 찾기 (No. 헤더 아래 숫자로 시작하는 행)
        product_rows = []
        for r in range(header_row + 1, header_row + 30):
            v = ws.cell(r, no_cell.column).value
            if v is None:
                break
            try:
                int(v)
                product_rows.append(r)
            except (TypeError, ValueError):
                break

        # 제품 행 채우기 (최대 기존 행 수만큼, 넘치면 마지막 행에 병합)
        for i, prod in enumerate(제품목록):
            if i >= len(product_rows):
                break
            r = product_rows[i]
            ws.cell(r, no_cell.column).value = i + 1
            ws.cell(r, no_cell.column + 1).value = prod.get("po_no", "")
            ws.cell(r, no_cell.column + 2).value = prod.get("brand", "")
            ws.cell(r, no_cell.column + 3).value = prod.get("description", "")
            ws.cell(r, no_cell.column + 4).value = prod.get("product_name_kr", "")
            ws.cell(r, no_cell.column + 5).value = prod.get("barcode", "")
            ws.cell(r, no_cell.column + 6).value = prod.get("qty", 0)
            ws.cell(r, no_cell.column + 7).value = prod.get("unit_price", 0)
            qty = int(prod.get("qty", 0))
            unit = float(prod.get("unit_price", 0))
            ws.cell(r, no_cell.column + 8).value = qty * unit
            ws.cell(r, no_cell.column + 9).value = prod.get("already_paid", "")
            ws.cell(r, no_cell.column + 10).value = ""  # Balance (TOTAL에서 계산)

        # 나머지 기존 행 비우기
        for r in product_rows[len(제품목록):]:
            for col in range(no_cell.column, no_cell.column + 11):
                ws.cell(r, col).value = None

        # TOTAL 행 업데이트
        total_cell = self._find_cell(ws, "TOTAL")
        if total_cell:
            total_row = total_cell.row
            total_qty = sum(int(p.get("qty", 0)) for p in 제품목록)
            ws.cell(total_row, no_cell.column + 6).value = total_qty
            ws.cell(total_row, no_cell.column + 8).value = 총액
            ws.cell(total_row, no_cell.column + 9).value = 기납부액

    def _fill_deposits(self, ws, 디포짓_목록: list[dict], 총액: float):
        """1st Deposit 행부터 채움. 기본 2차, 최대 4차."""
        LABELS = ["1st Deposit", "2st Deposit", "3st Deposit", "4st Deposit", "Blance"]
        for i, label in enumerate(LABELS[:4]):
            c = self._find_cell(ws, label)
            if c is None:
                continue
            if i < len(디포짓_목록):
                dep = 디포짓_목록[i]
                pct = float(dep.get("pct", 0))
                amt = float(dep.get("amount") or (총액 * pct))
                ws.cell(c.row, c.column + 1).value = round(pct, 4)
                ws.cell(c.row, c.column + 2).value = round(amt, 2)
            else:
                # 사용 안 하는 차수는 0으로
                ws.cell(c.row, c.column + 1).value = 0
                ws.cell(c.row, c.column + 2).value = 0

        # Balance 행
        balance_c = self._find_cell(ws, "Blance")
        if balance_c:
            총_디포짓 = sum(
                float(d.get("amount") or (총액 * float(d.get("pct", 0))))
                for d in 디포짓_목록
            )
            잔액 = round(총액 - 총_디포짓, 2)
            ws.cell(balance_c.row, balance_c.column + 1).value = 0
            ws.cell(balance_c.row, balance_c.column + 2).value = max(잔액, 0)

    def _fill_bank_info(self, ws, 계좌: dict):
        """Bank information 블록 셀 업데이트."""
        _MAP = [
            ("County / City",                 lambda c: f"{c.get('country','')} / {c.get('city','')}"),
            ("Beneficiary Account Number",     lambda c: c.get("account_number", "")),
            ("Beneficiary Name",               lambda c: c.get("beneficiary_name", "")),
            ("Beneficiary Address",            lambda c: c.get("beneficiary_address", "")),
            ("SWIFT/BIC code",                 lambda c: c.get("swift", "")),
            ("Bank name",                      lambda c: c.get("bank_name", "")),
            ("Bank Address",                   lambda c: c.get("bank_address", "")),
            ("Branch Code",                    lambda c: c.get("branch_code", "")),
        ]
        for keyword, getter in _MAP:
            cell = self._find_cell(ws, keyword)
            if cell:
                # 값은 같은 행 D열(keyword 다음 셀 또는 고정 열)에 있음
                # 템플릿 패턴: "keyword : value" → 같은 셀에 있거나 옆 셀
                현재값 = str(cell.value or "")
                if ":" in 현재값:
                    # "keyword : 기존값" 형식이면 키워드 부분만 남기고 값 교체
                    prefix = 현재값.split(":")[0] + ": "
                    cell.value = prefix + getter(계좌)
                else:
                    # 다음 셀에 값이 있는 경우
                    ws.cell(cell.row, cell.column + 1).value = getter(계좌)

    def _insert_stamps(self, ws, 계좌: dict):
        """누아트 도장 + 업체 도장 삽입."""
        # 누아트 도장 (Buyer 서명란)
        if config.누아트_도장.exists():
            buyer_cell = self._find_cell(ws, "Buyer")
            if buyer_cell:
                img = XLImage(str(config.누아트_도장))
                img.width, img.height = 80, 50
                col_letter = openpyxl.utils.get_column_letter(buyer_cell.column)
                ws.add_image(img, f"{col_letter}{buyer_cell.row + 1}")

        # 업체 도장 (Representative 서명란)
        도장_파일 = 계좌.get("도장_파일", "")
        if 도장_파일:
            도장_경로 = config.BASE_DIR / "assets" / 도장_파일
            if 도장_경로.exists():
                rep_cell = self._find_cell(ws, "Representative")
                if rep_cell:
                    img = XLImage(str(도장_경로))
                    img.width, img.height = 80, 50
                    col_letter = openpyxl.utils.get_column_letter(rep_cell.column)
                    ws.add_image(img, f"{col_letter}{rep_cell.row + 1}")
```

```python
"""
Slack Webhook 알림 발송
"""
import json
import config

try:
    import requests as _requests
except ImportError:
    _requests = None


def 슬랙_발송(메시지: str) -> bool:
    """Webhook으로 메시지 발송. 성공 True, 실패/미설정 False."""
    if not config.SLACK_WEBHOOK_URL or _requests is None:
        print(f"[슬랙] 미설정 또는 requests 없음. 메시지: {메시지}")
        return False
    try:
        r = _requests.post(
            config.SLACK_WEBHOOK_URL,
            data=json.dumps({"text": 메시지}),
            headers={"Content-Type": "application/json"},
            timeout=5,
        )
        return r.status_code == 200
    except Exception as e:
        print(f"[슬랙] 발송 실패: {e}")
        return False


def 담당자_멘션(이름: str) -> str:
    """이름 → <@U12345> 형식. 맵에 없으면 이름 그대로."""
    uid = config.SLACK_담당자_맵.get(이름, "")
    return f"<@{uid}>" if uid else 이름


def pi_알림_발송(담당자: str, po코드: str, 차수: str, pct: float, 금액: float) -> bool:
    """
    예: @홍길동 [NFA001] 1차 30% 금일 송금해야합니다. (¥14,100)
    """
    멘션 = 담당자_멘션(담당자)
    메시지 = (
        f"{멘션} [{po코드}] {차수} {int(pct*100)}% "
        f"금일 송금해야합니다. (¥{금액:,.0f})"
    )
    return 슬랙_발송(메시지)
```

```python
    def PI_완료_처리(self, po코드: str, 디포짓_인덱스: int) -> dict:

    # ── 대시보드 조회 ─────────────────────────────────────────────

    def 담당자_PI_목록(self, 담당자: str, sheets=None) -> list[dict]:
        """
        담당자의 미완료 디포짓이 있는 PI 목록.
        엔대시 시트에서 누아트입고일자(K열) 조회하여 입고 상태 보강.

            if meta.get("담당자") != 담당자:
                continue
            미완료_deps = [

                if not d.get("완료")

            if not 미완료_deps:
                continue

            # 제품별 입고 상태 (바코드 기준)

                "담당자":    meta["담당자"],

                "생성일":    meta["생성일"],
                "입고상태":  입고상태,
                "제품목록":  meta["제품목록"],
                "디포짓":    미완료_deps,
            })

        return sorted(결과, key=lambda x: x["생성일"], reverse=True)

    def 전체_PI_목록(self) -> list[dict]:
        """전체 PI 요약 목록 (대시보드용)."""

    # ── 슬랙 스케줄러 (매일 호출) ─────────────────────────────────

    def 오늘_알림_발송(self) -> int:
        """오늘 예정일인 미완료 디포짓에 슬랙 알림. 발송 건수 반환."""
        today = date.today().isoformat()
        db = self._db_읽기()

        for meta in db.values():
            for dep in meta.get("디포짓", []):

                    continue
                if dep.get("예정일", "") == today:

                        float(dep.get("pct", 0)),
                        float(dep.get("amount", 0)),
                    )
                    if ok:
                        count += 1

    def _db_읽기(self) -> dict:
        if not self._데이터_파일.exists():
            return {}
        return json.loads(self._데이터_파일.read_text(encoding="utf-8"))

    def _db_저장(self, db: dict):
        self._데이터_파일.write_text(
            json.dumps(db, ensure_ascii=False, indent=2), encoding="utf-8"
        )

    def _입고일자_조회(self, sheets) -> dict[str, str]:
        """엔대시 시트 K열(누아트입고일자)을 po코드 기준으로 읽어옴."""
        if not sheets:
            return {}
        try:
            result = sheets._service.spreadsheets().values().get(
                spreadsheetId=sheets._스프레드시트_id,
                range=f"'{sheets._시트명}'!B:K",  # B=바코드, C=발주번호, K=누아트입고일자
            ).execute()
            rows = result.get("values", [])
            맵 = {}
            for row in rows[1:]:
                if len(row) < 11:
                    continue
                po = str(row[2]).strip() if len(row) > 2 else ""
                입고일 = str(row[10]).strip() if len(row) > 10 else ""
                if po and 입고일:
                    맵[po] = 입고일
            return 맵
        except Exception as e:
            print(f"[PI유스케이스] 입고일자 조회 실패: {e}")
            return {}

    def _입고상태_판단(self, 바코드들: list, 입고맵: dict, po코드: str) -> str:
        입고일 = 입고맵.get(po코드, "")
        if not 입고일:
            return "입고 미정"
        try:
            d = datetime.strptime(입고일.replace(". ", "-").replace(".", "").strip(), "%Y-%m-%d").date()
            if d <= date.today():
                return f"입고 완료 ({입고일})"
            return f"입고 예정 {입고일}"
        except ValueError:
            return f"입고 예정 {입고일}"
```

```html
async function downloadPI(po){
  const r = await fetch('/api/pi/목록').then(r=>r.json());
  const pi = (r.data||[]).find(p=>p.po코드===po);
  if(!pi||!pi.파일경로) return alert('파일 없음');
  // 절대경로에서 output/ 이하 상대경로 추출

}
```

```javascript
// 계좌 DB에서 제조사명 자동 매칭
let _acctDB = [];
async function loadAcctDB(){
  const r = await fetch('/api/계좌/목록').then(r=>r.json());
  _acctDB = r.data||[];
  const sel = document.getElementById('acct-link-sel');
  if(sel) sel.innerHTML = '<option value="">— 수동 선택 —</option>' +
    _acctDB.map(a=>`<option value="${a.id}">${a.회사명_영문}</option>`).join('');
}
function onMakerInput(val){
  const row = document.getElementById('acct-link-row');
  if(!row) return;
  row.style.display = val ? 'block' : 'none';
  const matched = _acctDB.find(a=>
    a.회사명_영문.toLowerCase().includes(val.toLowerCase()) ||
    (a.회사명_중문||'').includes(val)
  );
  document.getElementById('acct-matched-name').textContent = matched ? '→ '+matched.회사명_영문 : '';
  if(matched) document.getElementById('acct-link-sel').value = matched.id;
}
function toggleAcctLink(){
  const chk = document.getElementById('acct-link-chk').checked;
  document.getElementById('acct-link-sel').style.display = chk ? 'inline-block' : 'none';
}
// 페이지 로드 시 계좌 DB 로드
document.addEventListener('DOMContentLoaded', loadAcctDB);
```
